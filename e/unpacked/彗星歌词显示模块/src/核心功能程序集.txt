CopyMemory_Rect (Rect, lParam, 16)
x = Rect.Left
y = Rect.Top
w = Rect.Right - Rect.Left
h = Rect.Bottom - Rect.Top
SystemParametersInfo (#SPI_GETWORKAREA, 0, WorkArea, 0)  ' //先取出桌面工作区的矩形位置
.如果真 (x < WorkArea.Left)
    x = WorkArea.Left
.如果真结束
.如果真 (y < WorkArea.Top)
    y = WorkArea.Top
.如果真结束
.如果真 (x + w > WorkArea.Right)
    x = WorkArea.Right - w
.如果真结束
.如果真 (y + h > WorkArea.Bottom)
    y = WorkArea.Bottom - h
.如果真结束
.如果真 (x != Rect.Left || y != Rect.Top)
    Rect = CRect (x, y, x + w, y + h)
    CopyMemory_Rect2 (lParam, Rect, 16)
.如果真结束
返回 (DefWindowProc (hWnd, wMsg, wParam, lParam))
